Imagens de Retina

  1. Use duas vesões da imagem a original e outra com o dobro do tamanho
  2.   
/* normal - 12px por 16px */

.button {
  background-image: url(images/button.png) no-repeat;
}

/* retina */

@media
only screen and (-webkit-min-device-pixel-ration: 1.5),
only screen and (min-device-pixel-ration: 1.5) {
  .button {
    background-image: url(images/button@2x.png); //imagem 24px por 32px
    /* seta o background size para o tamanho da imagem original */
    -webkit-background-size: 12px 16px;
    background-size: 12px 16px;
  }
}
